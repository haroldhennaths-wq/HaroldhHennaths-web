La API key vive SOLO aquí (variable de entorno en Netlify), // nunca en el navegador del usuario. const GROQ_URL = "https://api.groq.com/openai/v1/chat/completions"; const MODEL = "llama-3.3-70b-versatile"; // rápido, gratis (con límites) y muy capaz // ---- Contexto/identidad del agente ---------------------------------------- // Aquí le damos al modelo el conocimiento real de tu negocio para que responda // de forma libre pero siempre alineado a lo que ofreces. const SYSTEM_PROMPT = ` Eres "Harold IA", el asistente virtual del sitio web de Harold Hennaths, un Growth Partner que implementa Inteligencia Artificial y automatización para negocios (restaurantes, clínicas, spas, peluquerías, gimnasios, inmobiliarias, empresas B2B, educación, etc.). body.history.slice(-10) : []; if (!userMessage.trim()) { return { statusCode: 400, headers, body: JSON.stringify({ error: "Mensaje vacío" }) }; } const messages = [ { role: "system", content: SYSTEM_PROMPT }, ...history.filter( (m) => m && (m.role === "user" || m.role === "assistant") && typeof m.content === "string" ), { role: "user", content: userMessage }, ]; const response = await fetch(GROQ_URL, { method: "POST", headers: { "Content-Type": "application/json", Authorization: `Bearer ${process.env.GROQ_API_KEY}`, }, body: JSON.stringify({ model: MODEL, messages, temperature: 0.7, max_tokens: 500, }), }); if (!response.ok) { const errText = await response.text(); console.error("Groq error:", response.status, errText); return { statusCode: 502, headers, body: JSON.stringify({ error: "Error al contactar el modelo de IA.
